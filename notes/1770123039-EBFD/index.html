0.1 · Standard library: context

Package context defines the Context type, which carries deadlines, cancellation signals, and other request-scoped values across API boundaries and between processes.

context 包定义了一个上下文Context类型，这个类型在跨越 API 边界和进程之间，携带了超时，取消信号，以及其他请求域内的值。 也就是说，在不同函数（goroutine也是一个函数）之间传递Context值，可以实现超时，取消功能，也可以实现跨越函数的参数传递。

0.1.1 · 总览

根据context包的文档，可以知道：

使用 func Background() Context 或 func TODO() Context 可以创建一个空白的Context，不能被取消、没有截至时间、不携带值…

然后可以使用 WithCancel, WithDeadline, WithTimeout, WithValue 这几个函数接受一个Context参数，派生出一个新的Context，这个新的Context可以被取消、有截至时间、携带值…

WithCancel, WithDeadline, WithTimeout 这三个函数，接受一个Context参数，派生出一个新的Context与一个 type CancelFunc func() 函数，调用CancelFunc函数，这个派生出来的Context在取消的同时，也会取消由这个Context派生出的所有Context，如果有定时器，也会被取消。

WithValue 函数，接受一个Context参数与一对Key: Value，返回一个新的Context，这个新的Context携带了这个Key和Value。

0.1.2 · 例子

0.1.2.1 · WithCancel: Playground

package main

import (
	"context"
	"fmt"
)

func main() {
    // 使用 WithCancel 创建 Context 和 CancelFunc
	ctx, cancel := context.WithCancel(context.Background())
    // 在函数 main 返回后，调用CancelFunc取消 Context
	defer cancel()

    // gen函数返回一个<-chan int，同时创建一个协程不断往这个<-chan上发送了值
    // 当Context的CancelFunc被调用时，Context.Done()不再被阻塞，从而退出了gen函数
	gen := func(ctx context.Context) <-chan int {
		dst := make(chan int)
		n := 1
		go func() {
			for {
				select {
				case <-ctx.Done():
					return // 返回，不会导致协程泄露
				case dst <- n:
					n++
				}
			}
		}()
		return dst
	}

	for n := range gen(ctx) {
		fmt.Println(n)
		if n == 5 {
			break
		}
	}
}

0.1.2.2 · WithDeadline: Playground

package main

import (
	"context"
	"fmt"
	"time"
)

const shortDuration = 1 * time.Millisecond

func main() {
	d := time.Now().Add(shortDuration)
	// 使用 WithDeadline 创建 Context 和 CancelFunc
	ctx, cancel := context.WithDeadline(context.Background(), d)
	// 即使这个Context将会过期，也要调用CancelFunc函数，以确保Context和它的父Context永远不会被长时间保持。
	defer cancel()

	select {
	case <-time.After(1 * time.Second):
		fmt.Println("overslept")
	case <-ctx.Done(): // Context过期时，Context.Done()不再被阻塞
		fmt.Println(ctx.Err())
	}
}

0.1.2.4 · WithValue Playground

package main

import (
	"context"
	"fmt"
)

func main() {
	type favContextKey string

	k := favContextKey("language")

	// 使用 WithValue 创建携带值的 Context: key=k, value=Go
	ctx := context.WithValue(context.Background(), k, "Go")

	f := func(ctx context.Context, k favContextKey) {
		// 使用ctx.Value获取值
		if v := ctx.Value(k); v != nil {
			fmt.Println("found value:", v)
			return
		}
		fmt.Println("key not found:", k)
	}

	f(ctx, k)                      // 获取key=k的值，打印: found value: Go
	f(ctx, favContextKey("color")) // 获取key=favContextKey("color")的值，打印: key not found: color
}

0.1.3 · 实现

0.1.3.1 · 核心：type Context interface{}

// 接口Context定义了上下文的实现需要有哪些行为
type Context interface {
    // Deadline returns the time when this Context will be canceled, if any.
	// 返回Context的截止时间
    Deadline() (deadline time.Time, ok bool)
    // Done returns a channel that is closed when this Context is canceled
    // or times out.
	// 返回一个通道，当Context被取消或者超时的时候，这个通道会被关闭
    Done() <-chan struct{}
    // Err indicates why this context was canceled, after the Done channel
    // is closed.
	// 返回Context的取消原因，在Done通道被关闭之后
    Err() error
    // Value returns the value associated with key or nil if none.
	// 返回key对应的值，如果没有返回nil
    Value(key interface{}) interface{}
}

0.1.3.2 · type Context interface 的实现

// emptyCtx实现了接口Context
// 其不可被取消，没有值，没有deadline
type emptyCtx int

func (*emptyCtx) Deadline() (deadline time.Time, ok bool) {
	return
}

func (*emptyCtx) Done() <-chan struct{} {
	return nil
}

func (*emptyCtx) Err() error {
	return nil
}

func (*emptyCtx) Value(key any) any {
	return nil
}

func (e *emptyCtx) String() string {
	switch e {
	case background:
		return "context.Background"
	case todo:
		return "context.TODO"
	}
	return "unknown empty Context"
}

// 包内部定义的全局变量，外部包不可见
// 可以通过 context.Background()` 或 `context.TODO() 获取
var (
	background = new(emptyCtx)
	todo       = new(emptyCtx)
)

// 通常，我们会使用 context.Background() 来派生一个新的Context
func Background() Context {
	return background
}

func TODO() Context {
	return todo
}

0.1.3.3 · WithCancel

// 看WithCancel`函数签名，可知，接收一个parent Context作为参数，返回一个Context和一个CancelFunc；
func WithCancel(parent Context) (ctx Context, cancel CancelFunc) {
	if parent == nil { // 1. 先判断parent Context是否为nil，如果是，则panic；
		panic("cannot create context from nil parent")
	}
	c := newCancelCtx(parent) // 2. 调用newCancelCtx`函数，创建一个`cancelCtx结构体值；
	propagateCancel(parent, &c) // 3. 调用propagateCancel`函数，作用是当`parent Context`被取消时，所有的`child Context也会被取消；
	return &c, func() { c.cancel(true, Canceled) } // Canceled` 是包级别的err变量，表示`Context被取消了
}

先看第2步的newCancelCtx函数：

// newCancelCtx returns an initialized cancelCtx.
func newCancelCtx(parent Context) cancelCtx {
	return cancelCtx{Context: parent} // 初始化一个cancelCtx结构体值
}

type canceler interface {
	cancel(removeFromParent bool, err error)
	Done() <-chan struct{}
}

// cancelCtx`结构体有 `cancel` `Done` 两个方法，即cancelCtx实现了`canceler接口
type cancelCtx struct {
	Context // 内嵌接口，可视为cancelCtx实现了Context接口

	mu       sync.Mutex            // protects following fields
	done     atomic.Value          // of chan struct{}, created lazily, closed by first cancel call
	children map[canceler]struct{} // set to nil by the first cancel call
	err      error                 // set to non-nil by the first cancel call
}

// 在WithCancel`函数内，返回了一个`Canceled`函数，调用 `cancel 方法
func (c *cancelCtx) cancel(removeFromParent bool, err error) {
	if err == nil {
		panic("context: internal error: missing cancel error")
	}
	// 对c.err字段赋值，如果已经赋值，则Context已经被取消，直接返回
	c.mu.Lock()
	if c.err != nil {
		c.mu.Unlock()
		return // already canceled
	}
	c.err = err

	// 获取c.done字段，储存的是chan struct{}, 存在则关闭，不存在则储存
	d, _ := c.done.Load().(chan struct{})
	if d == nil {
		c.done.Store(closedchan) // closedchan 是包级别声明的 chan struct{}，可被复用
	} else {
		close(d) // Done`函数返回的`chan struct{}不再阻塞
	}

	// 取消child Context
	for child := range c.children {
		// NOTE: acquiring the child's lock while holding parent's lock.
		child.cancel(false, err)
	}
	c.children = nil
	c.mu.Unlock()

	if removeFromParent {
		removeChild(c.Context, c) // 从parent Context`中移除`child Context
	}
}

// Done`函数，返回一个`chan struct{}`，当`Context`被取消时，`chan struct{}会被关闭
func (c *cancelCtx) Done() <-chan struct{} {
	d := c.done.Load()
	if d != nil {
		return d.(chan struct{})
	}
	// double-checked locking
	// 可能另一个协程拿到Lock了，这里被阻塞
	c.mu.Lock()
	defer c.mu.Unlock()
	// 此协程拿到Lock时，d.done字段可能已经被赋值的，所以需要再次判断
	d = c.done.Load()
	if d == nil {
		d = make(chan struct{})
		c.done.Store(d)
	}
	return d.(chan struct{})
}

// 取消Context时，返回的错误
func (c *cancelCtx) Err() error {
	c.mu.Lock()
	err := c.err
	c.mu.Unlock()
	return err
}

// 从parent Context`中移除`child Context
func removeChild(parent Context, child canceler) {
	// parentCancelCtx`函数返回`parent Context`的底层`*cancelCtx
	// 不是*cancelCtx直接返回
	p, ok := parentCancelCtx(parent)
	if !ok {
		return
	}
	// 否则，删除child Context
	p.mu.Lock()
	if p.children != nil {
		delete(p.children, child)
	}
	p.mu.Unlock()
}

然后看第3步的propagateCancel函数： 其作用是在parent Context被取消时，取消所有的child Context；如果parent Context未被取消时，绑定child Context；


type canceler interface {
	cancel(removeFromParent bool, err error)
	Done() <-chan struct{}
}

func propagateCancel(parent Context, child canceler) {
	// 返回 <-chan struct{}，如果为nil，说明parent不可被取消
	done := parent.Done()
	if done == nil {
		return // parent is never canceled
	}

	// 如果 parent Context` 已经被取消，则调用`child Context`的`cancel`方法取消`child Context
	// 然后直接返回
	select {
	case <-done:
		// parent is already canceled
		child.cancel(false, parent.Err())
		return
	default:
	}

	// parent Context 未被取消
	// parentCancelCtx`函数返回`parent Context`的底层`*cancelCtx
	// 底层*cancelCtx存在
	if p, ok := parentCancelCtx(parent); ok {
		p.mu.Lock()
		if p.err != nil { // parent Context已经被取消
			// parent has already been canceled
			child.cancel(false, p.err) // 取消child Context
		} else {
			// parent Context未被取消
			if p.children == nil {
				p.children = make(map[canceler]struct{})
			}
			p.children[child] = struct{}{} // 添加child Context`到`parent Context`的`children中
		}
		p.mu.Unlock()
	} else {
		// 底层*cancelCtx不存在
		atomic.AddInt32(&goroutines, +1)
		go func() {
			select {
			case <-parent.Done(): // parent Context取消
				child.cancel(false, parent.Err()) // 同时取消child Context
			case <-child.Done(): // 取消child Context
			}
		}()
	}
}

// parentCancelCtx`函数返回`parent Context`的底层`*cancelCtx
func parentCancelCtx(parent Context) (*cancelCtx, bool) {
	done := parent.Done()
	if done == closedchan || done == nil {
		return nil, false
	}
	p, ok := parent.Value(&cancelCtxKey).(*cancelCtx)
	if !ok {
		return nil, false
	}
	pdone, _ := p.done.Load().(chan struct{})
	if pdone != done {
		return nil, false
	}
	return p, true
}

0.1.3.4 · WithDeadline

func WithDeadline(parent Context, d time.Time) (Context, CancelFunc) {
	if parent == nil {
		panic("cannot create context from nil parent")
	}
	// 目前的截止日期已经比新的截止日期早，截止日期不改动
	if cur, ok := parent.Deadline(); ok && cur.Before(d) {
		// The current deadline is already sooner than the new one.
		return WithCancel(parent)
	}

	// 创建一个timerCtx结构体值
	c := &timerCtx{
		cancelCtx: newCancelCtx(parent),
		deadline:  d,
	}

	// 在parent Context`被取消时，取消所有的`child Context`；如果`parent Context`未被取消时，绑定`child Context；
	propagateCancel(parent, c)

	// 判断是否已经超时，若已经超时，则取消child Context
	dur := time.Until(d)
	if dur <= 0 {
		c.cancel(true, DeadlineExceeded) // deadline has already passed
		return c, func() { c.cancel(false, Canceled) }
	}
	c.mu.Lock()
	defer c.mu.Unlock()
	if c.err == nil { // 未超时，设置定时器
		c.timer = time.AfterFunc(dur, func() {
			c.cancel(true, DeadlineExceeded)
		})
	}
	return c, func() { c.cancel(true, Canceled) }
}

type timerCtx struct {
	cancelCtx
	timer *time.Timer // Under cancelCtx.mu.

	deadline time.Time
}

func (c *timerCtx) Deadline() (deadline time.Time, ok bool) {
	return c.deadline, true
}

// 取消child Context`同时停止`timer
func (c *timerCtx) cancel(removeFromParent bool, err error) {
	c.cancelCtx.cancel(false, err)
	if removeFromParent {
		// Remove this timerCtx from its parent cancelCtx's children.
		removeChild(c.cancelCtx.Context, c)
	}
	c.mu.Lock()
	if c.timer != nil {
		c.timer.Stop()
		c.timer = nil
	}
	c.mu.Unlock()
}

// 即调用WithDeadline函数

0.1.3.6 · WithValue

func WithValue(parent Context, key, val any) Context {
	if parent == nil {
		panic("cannot create context from nil parent")
	}
	if key == nil {
		panic("nil key")
	}
	if !reflectlite.TypeOf(key).Comparable() {
		panic("key is not comparable")
	}
	return &valueCtx{parent, key, val}
}

type valueCtx struct {
	Context
	key, val any
}

func (c *valueCtx) Value(key any) any {
	// 如果key在当前Context中存在，则返回val
	if c.key == key {
		return c.val
	}
	return value(c.Context, key)
}

// 对Context接口值进行断言，根据不同类型，返回不同的Value
func value(c Context, key any) any {
	for {
		switch ctx := c.(type) {
		case *valueCtx:
			if key == ctx.key {
				return ctx.val
			}
			c = ctx.Context
		case *cancelCtx:
			if key == &cancelCtxKey {
				return c
			}
			c = ctx.Context
		case *timerCtx:
			if key == &cancelCtxKey {
				return &ctx.cancelCtx
			}
			c = ctx.Context
		case *emptyCtx:
			return nil
		default:
			return c.Value(key)
		}
	}
}
