0.1.1 · Mutex:

Acquire and Release

等待、饥饿、自旋（唤醒）、上锁

两种模式：普通模式，饥饿模式；普通模式有着更好的性能

  1. 普通模式：先进先出 唤醒的等待者g需要与新到来的g去竞争锁 新到来的g有优势：已经运行在cpu，且可以有很多个新到来的g，唤醒的等待着很难获取到锁 这种情况下，会把唤醒的等待者放到等待队列头 如果唤醒的等待着等待超过1ms，进入饥饿模式

  2. 饥饿模式：锁直接给等待队列头唤醒的g 新到来的g不会去尝试获取锁，将其放在等待队列尾 如果唤醒的等待者拿到锁，同时1）其是等待队列最后一个g 2）其等待小于1ms，退回普通模式

type Mutex struct {
    // state状态: int32，32bit 分为四部分：
    // 00000000000000000000000000000(29bit, 等待者数量) 
    // 0(1bit, Mutex是否处于饥饿模式) 
    // 0(1bit, 申请Mutex的g是否唤醒) 
    // 0(1bit, Mutex是否被锁)`
	state int32
    // sema信号量: 唤醒g或让g等待
	sema  uint32
}

// 主要是Mutex的state状态的变化，涉及位操作：
const (
	mutexLocked = 1 << iota // mutex is locked  // 1
	mutexWoken // mutex has been woken  // 2
	mutexStarving // mutex is starving  // 4
	mutexWaiterShift = iota // number of waiters  // 3

	starvationThresholdNs = 1e6 // 1ms
)

func (m *Mutex) lockSlow() {
	var waitStartTime int64
	starving := false
	awoke := false
	iter := 0
	old := m.state
	for {
		// Don't spin in starvation mode, ownership is handed off to waiters
		// so we won't be able to acquire the mutex anyway.
        // old状态，锁被占有、无饥饿 且 g自旋有意义
        // 当前g设置m.state有唤醒 => 设置当前g唤醒 ;  
        // 进入自旋; 更新下方法内复制的old状态; continue跳回for循环
		if old&(mutexLocked|mutexStarving) == mutexLocked && runtime_canSpin(iter) {
			// Active spinning makes sense.
			// Try to set mutexWoken flag to inform Unlock
			// to not wake other blocked goroutines.
			// 当前g未唤醒 且old状态无唤醒的g 且old状态等待的g数不为0 且old状态未被其他g修改
			if !awoke && old&mutexWoken == 0 && old>>mutexWaiterShift != 0 && atomic.CompareAndSwapInt32(&m.state, old, old|mutexWoken) {
				awoke = true
			}
			runtime_doSpin()
			iter++
			old = m.state
			continue
		}
        // 复制old状态到new状态
		new := old
		// Don't try to acquire starving mutex, new arriving goroutines must queue.
		// 如果old状态没有饥饿的g，对new状态尝试上锁
		if old&mutexStarving == 0 {
			new |= mutexLocked
		}
		// 如果old状态有饥饿的g或锁被占有，new状态等待数量加1
		if old&(mutexLocked|mutexStarving) != 0 {
			new += 1 << mutexWaiterShift
		}
		// The current goroutine switches mutex to starvation mode.
		// But if the mutex is currently unlocked, don't do the switch.
		// Unlock expects that starving mutex has waiters, which will not
		// be true in this case.
		// 如果当前g为饥饿状态 且old状态锁被占有，对new状态设置有饥饿g
		if starving && old&mutexLocked != 0 {
			new |= mutexStarving
		}
		// 如果当前g被唤醒，清除new状态的有唤醒g标志(如果new状态的有唤醒g标志为0，则报错mutex状态不连续)
		if awoke {
			// The goroutine has been woken from sleep,
			// so we need to reset the flag in either case.
			if new&mutexWoken == 0 {
				throw("sync: inconsistent mutex state")
			}
			new &^= mutexWoken
		}
		// old状态未被其他g修改且更新成new状态成功		
		if atomic.CompareAndSwapInt32(&m.state, old, new) {
			// (如果old状态无饥饿且无饥饿g，跳出for循环，也就是成功获取锁；
			if old&(mutexLocked|mutexStarving) == 0 {
				break // locked the mutex with CAS
			}
			// If we were already waiting before, queue at the front of the queue.
			// 当前g的等待开始时间不为0
			queueLifo := waitStartTime != 0
			if waitStartTime == 0 {
				// 否则初始化等待开始时间
				waitStartTime = runtime_nanotime()
			}
			// 如果queueLifo等于true，则把当前g放到等待队列队首
			runtime_SemacquireMutex(&m.sema, queueLifo, 1)
			// 如果当前g饥饿或当前时间减去开始等待时间大于1ms，标记当前g为饥饿；
			starving = starving || runtime_nanotime()-waitStartTime > starvationThresholdNs
			// 复制mutex的状态到old；
			old = m.state
			// 如果old为饥饿状态
			if old&mutexStarving != 0 {
				// If this goroutine was woken and mutex is in starvation mode,
				// ownership was handed off to us but mutex is in somewhat
				// inconsistent state: mutexLocked is not set and we are still
				// accounted as waiter. Fix that.
				// 如果old状态有唤醒g或锁被占有 或old状态等待数量为0，报错
				if old&(mutexLocked|mutexWoken) != 0 || old>>mutexWaiterShift == 0 {
					throw("sync: inconsistent mutex state")
				}
				// 上锁且等待数量减1
				delta := int32(mutexLocked - 1<<mutexWaiterShift)
				// 如果当前g不饥饿或old状态的等待数为1
				if !starving || old>>mutexWaiterShift == 1 {
					// Exit starvation mode.
					// Critical to do it here and consider wait time.
					// Starvation mode is so inefficient, that two goroutines
					// can go lock-step infinitely once they switch mutex
					// to starvation mode.
					// 退出饥饿模式
					delta -= mutexStarving
				}
				// 更新mutex状态
				atomic.AddInt32(&m.state, delta)
				// 退出for循环
				break
			}
			// 设置当前g唤醒，重置自旋参数
			awoke = true
			iter = 0
		} else {
			// 否则其他g更新的状态，保存到当前g的old状态
			old = m.state
		}
	}

	if race.Enabled {
		race.Acquire(unsafe.Pointer(m))
	}
}

Lock：

  1. 此锁无等待g、非饥饿模式、无唤醒g、无g占有锁，当前g占有锁
  2. 进入循环模式，可能多个g操作此锁，先保存当前的state为old 2.1. old状态非饥饿模式、锁被其他g占有，同时当前g可以自旋，则当前g自旋，如果old无唤醒g，则修改锁state为有唤醒g； 2.2. old状态非饥饿模式，修改new状态加锁；old状态饥饿模式或被其他g占有，new状态等待者加1；当前g饥饿且old状态被占有，修改new状态为饥饿；当前g唤醒，修改new状态为无唤醒g；如果old未被其他g修改，则更新new状态到state； 假如更新成功，如果old状态无饥饿且无饥饿g，跳出for循环，也就是成功获取锁； 2.3 否则，更新state到old；重新从2.1开始；
func (m *Mutex) Unlock() {
	if race.Enabled {
		_ = m.state
		race.Release(unsafe.Pointer(m))
	}

	// Fast path: drop lock bit.
	// 释放锁，此时状态赋值为new
	new := atomic.AddInt32(&m.state, -mutexLocked)
	// 释放后state等于0则锁释放成功，
	if new != 0 {
		// Outlined slow path to allow inlining the fast path.
		// To hide unlockSlow during tracing we skip one extra frame when tracing GoUnblock.
		m.unlockSlow(new)
	}
}

func (m *Mutex) unlockSlow(new int32) {
	// 重新加锁，占有锁后的状态仍为0，报错
	if (new+mutexLocked)&mutexLocked == 0 {
		fatal("sync: unlock of unlocked mutex")
	}
	// new状态非饥饿模式
	if new&mutexStarving == 0 {
		// 保存new状态到old
		old := new
		// 开启for循环
		for {
			// If there are no waiters or a goroutine has already
			// been woken or grabbed the lock, no need to wake anyone.
			// In starvation mode ownership is directly handed off from unlocking
			// goroutine to the next waiter. We are not part of this chain,
			// since we did not observe mutexStarving when we unlocked the mutex above.
			// So get off the way.
			// 如果old状态等待者为0 或old状态锁被占有或有唤醒的g或有饥饿的g
			if old>>mutexWaiterShift == 0 || old&(mutexLocked|mutexWoken|mutexStarving) != 0 {
				// 直接返回不用释放锁
				return
			}
			// Grab the right to wake someone.
			// old状态减去一个等待着且尝试加上有唤醒g标志 赋值为new状态
			new = (old - 1<<mutexWaiterShift) | mutexWoken
			// 如果设置new状态成功，释放信号量且直接返回；
			if atomic.CompareAndSwapInt32(&m.state, old, new) {
				runtime_Semrelease(&m.sema, false, 1)
				return
			}
			// 更新锁的状态到old
			old = m.state
		}
	} else {
		// Starving mode: handoff mutex ownership to the next waiter, and yield
		// our time slice so that the next waiter can start to run immediately.
		// Note: mutexLocked is not set, the waiter will set it after wakeup.
		// But mutex is still considered locked if mutexStarving is set,
		// so new coming goroutines won't acquire it.
		// 饥饿模式，直接把锁给等待队列的队首
		// 增加信号量
		// 唤醒等待的g
		// 如果第二个参数为true，直接传递给第一个等待g
		runtime_Semrelease(&m.sema, true, 1)
	}
}

func (m *Mutex) TryLock() bool {
	old := m.state
	if old&(mutexLocked|mutexStarving) != 0 {
		return false
	}

	// There may be a goroutine waiting for the mutex, but we are
	// running now and can try to grab the mutex before that
	// goroutine wakes up.
	if !atomic.CompareAndSwapInt32(&m.state, old, old|mutexLocked) {
		return false
	}

	if race.Enabled {
		race.Acquire(unsafe.Pointer(m))
	}
	return true
}

0.1.2 · RWMutex

某一时刻只能由任意数量的reader持有或被单个的writer持有。

Lock/UnLock 写操作调用； RLock/RUnlock 读操作调用； RLocker 返回一个Locker接口的对象，Lock/UnLock 进行读操作

场景明确却分R/W g, 有大量的并发读，少量的并发写，考虑RWMutex。

基于对写、读操作的优先级，设计与实现分为三类：

  1. 读优先，可提供很高并发性，但是在竞争激烈情况下可能导致写饥饿(大量的读，只有在所有的读都释放锁后，写才可以获得锁)
  2. 写优先，当有写在阻塞时，会阻止新来的读获取到锁，优先保障写，但是新来的写前有读获取到锁，那还是需要等待读释放锁
  3. 不指定优先，1可能导致写饥饿，2可能导致读饥饿

Go stdlib 中的设计是2，写优先，

0.1.3 · WaitGroup

解决并发-等待问题： gA阻塞在检查点，等待一组g完成任务。

type WaitGroup struct {
	// 高32位是计数器，低32位是等待计数
	// 64位的原子操作要求64位对齐，但32位编译器只能保证64位字段是32位对齐。
	// 在32位操作系统上，我们需要在state()中检查state1是否对齐，如果不对齐，则需要动态地交换字段顺序。
	// 剩下的32位做信号量
	state1 uint64
	state2 uint32
}

// 在创建goroutine或其他事件之前，必须先执行Add
func (wg *WaitGroup) Add(delta int) {
	statep, semap := wg.state()
	// race，忽略
	if race.Enabled {
		_ = *statep // trigger nil deref early
		if delta < 0 {
			// Synchronize decrements with Wait.
			race.ReleaseMerge(unsafe.Pointer(wg))
		}
		race.Disable()
		defer race.Enable()
	}
	// 高32位添加delta
	state := atomic.AddUint64(statep, uint64(delta)<<32)

	// 高32位值
	v := int32(state >> 32)
	// 低32位值
	w := uint32(state)

	// race，忽略
	if race.Enabled && delta > 0 && v == int32(delta) {
		// The first increment must be synchronized with Wait.
		// Need to model this as a read, because there can be
		// several concurrent wg.counter transitions from 0.
		race.Read(unsafe.Pointer(semap))
	}
	if v < 0 {
		panic("sync: negative WaitGroup counter")
	}
	// 高32位值等于当前添加的delta，且等待数量不为0
	if w != 0 && delta > 0 && v == int32(delta) {
		panic("sync: WaitGroup misuse: Add called concurrently with Wait")
	}
	// 高32位值不为0且等待数量为0
	if v > 0 || w == 0 {
		return
	}
	// This goroutine has set counter to 0 when waiters > 0.
	// Now there can't be concurrent mutations of state:
	// - Adds must not happen concurrently with Wait,
	// - Wait does not increment waiters if it sees counter == 0.
	// Still do a cheap sanity check to detect WaitGroup misuse.
	if *statep != state {
		panic("sync: WaitGroup misuse: Add called concurrently with Wait")
	}
	// Reset waiters count to 0.
	*statep = 0
	for ; w != 0; w-- {
		runtime_Semrelease(semap, false, 0) // 唤醒
	}
}

// Wait blocks until the WaitGroup counter is zero.
func (wg *WaitGroup) Wait() {
	statep, semap := wg.state()
	// race，忽略
	if race.Enabled {
		_ = *statep // trigger nil deref early
		race.Disable()
	}
	for {
		state := atomic.LoadUint64(statep)
		v := int32(state >> 32)
		w := uint32(state)
		// 高32位值等于0
		if v == 0 {
			// Counter is 0, no need to wait.
			if race.Enabled {
				race.Enable()
				race.Acquire(unsafe.Pointer(wg))
			}
			return
		}
		// Increment waiters count.
		// 增加低32位值
		if atomic.CompareAndSwapUint64(statep, state, state+1) {
			if race.Enabled && w == 0 {
				// Wait must be synchronized with the first Add.
				// Need to model this is as a write to race with the read in Add.
				// As a consequence, can do the write only for the first waiter,
				// otherwise concurrent Waits will race with each other.
				race.Write(unsafe.Pointer(semap))
			}
			runtime_Semacquire(semap) // 阻塞
			if *statep != 0 {
				panic("sync: WaitGroup is reused before previous Wait has returned")
			}
			if race.Enabled {
				race.Enable()
				race.Acquire(unsafe.Pointer(wg))
			}
			return
		}
	}
}
